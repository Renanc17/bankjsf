CREATE SCHEMA bankjsf ;

CREATE TABLE bankjsf.cliente (
  id INT NOT NULL AUTO_INCREMENT,
  nome VARCHAR(45) NOT NULL,
  cpf INT NOT NULL,
  agencia INT NOT NULL,
  senha VARCHAR(45) NOT NULL,
  contac INT NOT NULL,
  contap INT NOT NULL,
  PRIMARY KEY (id));

CREATE TABLE bankjsf.contac (
  contacid INT NOT NULL,
  saldo DOUBLE NULL,
  PRIMARY KEY (contacid));

CREATE TABLE bankjsf.contap (
  contapid INT NOT NULL,
  saldo DOUBLE NULL,
  PRIMARY KEY (contapid));
