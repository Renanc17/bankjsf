CREATE SCHEMA bankjsf ;

CREATE TABLE bankjsf.cliente (
  id INT NOT NULL AUTO_INCREMENT,
  nome VARCHAR(45) NOT NULL,
  cpf bigint NOT NULL,
  agencia INT NOT NULL,
  senha VARCHAR(45) NOT NULL,
  contac INT NOT NULL,
  saldoc DOUBLE NULL,
  contap INT NOT NULL,
  saldop DOUBLE NULL,
  senhaCartao INT NOT NULL,
  PRIMARY KEY (id));
  
CREATE TABLE bankjsf.cartao (
  id INT NOT NULL,
  s1 VARCHAR(4) NOT NULL,
  s2 VARCHAR(4) NOT NULL,
  s3 VARCHAR(4) NOT NULL,
  s4 VARCHAR(4) NOT NULL,
  s5 VARCHAR(4) NOT NULL,
  PRIMARY KEY (id));
